							Result = Struct.new(:success, :value) do
								def success?
									success
								end
							end

							class ChainIt
								def chain
									return self if @skip
									@result = yield @result&.value
									@skip = true unless @result.success?
									self
								end

								def result
									@result
								end
						 end


							ChainIt.new.chain { Result.new(true) }.
                   chain { Result.new(false) }.
                   chain { Result.new(true) }
